    "def draw_bs_rep(data, func, rng):\n",
    "    \"\"\"Compute a bootstrap replicate from data.\"\"\"\n",
    "    bs_sample = rng.choice(data, size=len(data))\n",
    "    return func(bs_sample)\n",
    "\n",
    "\n",
    "# Replicates of each year\n",
    "n_reps = 2000\n",
    "bs_reps_1975 = np.array(\n",
    "    [draw_bs_rep(bd_1975, np.mean, rng) for _ in range(n_reps)]\n",
    ")\n",
    "bs_reps_2012 = np.array(\n",
    "    [draw_bs_rep(bd_2012, np.mean, rng) for _ in range(n_reps)]\n",
    ")\n",
    "\n",
    "# Get replicates of the difference\n",
    "bd_reps_diff = bs_reps_2012 - bs_reps_1975\n",
    "\n",
